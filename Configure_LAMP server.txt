- name: linux KK task 4_5
  hosts: apps
  become: true
  tasks:
  - name: Install the latest version of httpd
    ansible.builtin.yum:
      name: 
       - httpd
       - php
       - php-mysqlnd
      state: present
  - name: edit httpd port
    ansible.builtin.lineinfile:
      dest: /etc/httpd/conf/httpd.conf
      regexp: "^Listen 80" 
      line: "Listen 5000"
  - name: start and enable service httpd
    ansible.builtin.service:
      name: httpd
      enabled: yes
      state: started
- name: mariadb
  hosts: db
  become: true
  tasks:
# we need to copy epel gpg key to avoid gpg import question when installing epel repo - let's copy it from our jump server  
  - name: Import repo GPG key
    copy:
      src=/etc/pki/rpm-gpg/RPM-GPG-KEY-EPEL-8
      dest=/etc/pki/rpm-gpg/
      owner=root
      group=root
      mode=644
  - name: Install the latest version of mariadb
    ansible.builtin.yum:
      name: 
      - mariadb-server
      - epel-release
      state: present
  - name: start and enable service mariadb
    ansible.builtin.service:
      name: mariadb
      enabled: yes
      state: started    
  - name: Install MySQL-python
    yum:
      name:
      - python3-pip
      - python3-mysql
      enablerepo: "epel"
  - name: create new database
    mysql_db:
       name: kodekloud_db9 
       state: present
  - name: set mysql user pass
    mysql_user: 
       name: kodekloud_roy
       host '%'             # we have 3 app servers - it needs to able connect to base from outside 
       password: TmPcZjtRQx
       priv: 'kodekloud_db19.*:ALL'
       state: present
